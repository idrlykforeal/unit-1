def mystery(a:int,b:int)->int:
    result=a*b-abs(a-b)
    return result

out=mystery(2,6)
print(out)

out=mystery(4,10)
print(out)

out=mystery(10,10)
print(out)

out=mystery(70,50)
print(out)
